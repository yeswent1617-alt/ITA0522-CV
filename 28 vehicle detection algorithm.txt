import cv2

# 1. Load the classifier
car_cascade = cv2.CascadeClassifier(r'C:\Users\HP\Documents\openCV\cars.xml')

# 2. Open the video
cap = cv2.VideoCapture(r'C:\Users\HP\Documents\openCV\car.mp4')

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    # 3. Convert to grayscale
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    # 4. Detect with VERY STRICT parameters
    # Increase minNeighbors significantly to filter out weak "wheel" patterns
    # minSize must be large enough to skip a wheel but small enough to catch the car
    cars = car_cascade.detectMultiScale(
        gray, 
        scaleFactor=1.1, 
        minNeighbors=15,    # High value to ignore false positives
        minSize=(180, 180)  # Ignore anything smaller than this (Adjust based on car size)
    )

    # 5. Draw detection boxes
    for (x, y, w, h) in cars:
        cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 255, 0), 3)

    cv2.imshow('Car Only Detection', frame)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
