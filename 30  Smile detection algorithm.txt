import cv2

# 1. Load the pre-trained Haar Cascades
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
smile_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_smile.xml')

# 2. Read the image
img = cv2.imread(r'C:\Users\HP\Documents\openCV\face.jpeg')

# Check if image was actually loaded
if img is None:
    print("Error: Could not find the image. Check your file path!")
else:
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    # 3. Detect faces
    faces = face_cascade.detectMultiScale(gray, 1.3, 5)

    for (x, y, w, h) in faces:
        roi_gray = gray[y:y+h, x:x+w]
        roi_color = img[y:y+h, x:x+w]

        # 4. Detect smile (High minNeighbors to prevent false detections)
        smiles = smile_cascade.detectMultiScale(roi_gray, 1.8, 20)

        for (sx, sy, sw, sh) in smiles:
            cv2.rectangle(roi_color, (sx, sy), (sx + sw, sy + sh), (0, 0, 255), 2)

    # 5. Display and WAIT
    cv2.imshow('Smile Detection Output', img)
    
    print("Window is open. Press ANY KEY on your keyboard to close it.")
    
    # Wait indefinitely until a key is pressed
    cv2.waitKey(0) 
    cv2.destroyAllWindows()
